## 三、通用部署脚本模板

### 3.1 一键部署脚本（PowerShell）

创建 `deploy_model.ps1`：

```powershell
# ==================== 配置区 ====================
param(
    [string]$ModelName = "whisperx",      # 模型名称
    [string]$ModelRoot = "E:\ai-deploy\models",
    [string]$VenvPath = "E:\ai-deploy\$ModelName-env",
    [string]$HFToken = "",                # 可选：HF Token
    [string]$PythonVersion = "3.10"
)

# 颜色输出
function Write-Step { param($msg) Write-Host "`n[步骤] $msg" -ForegroundColor Cyan }
function Write-Success { param($msg) Write-Host "[成功] $msg" -ForegroundColor Green }
function Write-Error { param($msg) Write-Host "[错误] $msg" -ForegroundColor Red; exit 1 }

# ==================== 步骤1：环境检查 ====================
Write-Step "检查环境..."

# 检查 NVIDIA GPU
$nvidia = nvidia-smi 2>$null
if (-not $nvidia) { Write-Error "未检测到 NVIDIA GPU，请安装驱动" }
Write-Success "GPU 检测通过"

# 检查 Python
$python = python --version 2>$null
if (-not $python) { Write-Error "未检测到 Python" }
Write-Success "Python: $python"

# ==================== 步骤2：配置镜像 ====================
Write-Step "配置镜像源..."

# pip 镜像
pip config set global.index-url https://mirrors.aliyun.com/pypi/simple/ 2>$null

# HF 镜像
[Environment]::SetEnvironmentVariable("HF_ENDPOINT", "https://hf-mirror.com", "User")
$env:HF_ENDPOINT = "https://hf-mirror.com"

Write-Success "镜像配置完成"

# ==================== 步骤3：创建虚拟环境 ====================
Write-Step "创建虚拟环境: $VenvPath"

if (Test-Path $VenvPath) {
    Write-Host "虚拟环境已存在，跳过创建"
} else {
    python -m venv $VenvPath
    Write-Success "虚拟环境创建完成"
}

# 激活虚拟环境
& "$VenvPath\Scripts\Activate.ps1"
Write-Success "虚拟环境已激活"

# ==================== 步骤4：安装依赖 ====================
Write-Step "安装 PyTorch..."

pip install torch torchaudio --index-url https://download.pytorch.org/whl/cu128
if (-not $?) { Write-Error "PyTorch 安装失败" }

# 验证 PyTorch
python -c "import torch; assert torch.cuda.is_available(), 'CUDA 不可用'; print(f'PyTorch: {torch.__version__}')"
Write-Success "PyTorch 安装完成"

# ==================== 步骤5：安装模型包 ====================
Write-Step "安装 $ModelName..."

# 根据模型名称安装不同依赖
switch ($ModelName) {
    "whisperx" {
        pip install whisperx fastapi uvicorn python-multipart gradio pandas
    }
    "qwen-asr" {
        pip install transformers accelerate fastapi uvicorn gradio
    }
    default {
        Write-Error "未知模型: $ModelName"
    }
}

Write-Success "$ModelName 安装完成"

# ==================== 步骤6：创建目录 ====================
Write-Step "创建目录结构..."

New-Item -ItemType Directory -Force -Path $ModelRoot | Out-Null
New-Item -ItemType Directory -Force -Path "E:\ai-deploy\services" | Out-Null
New-Item -ItemType Directory -Force -Path "E:\ai-deploy\logs\$ModelName" | Out-Null

Write-Success "目录创建完成"

# ==================== 完成 ====================
Write-Host "`n========================================" -ForegroundColor Green
Write-Host "部署完成！" -ForegroundColor Green
Write-Host "========================================" -ForegroundColor Green
Write-Host "虚拟环境: $VenvPath"
Write-Host "模型目录: $ModelRoot"
Write-Host "服务目录: E:\ai-deploy\services"
Write-Host ""
Write-Host "下一步："
Write-Host "1. 激活环境: & '$VenvPath\Scripts\Activate.ps1'"
Write-Host "2. 下载模型: python download_models.py"
Write-Host "3. 启动服务: python server.py"
```

### 3.2 模型下载脚本模板

创建 `download_models.py`：

```python
"""
通用模型下载脚本模板
使用方法：修改 CONFIG 部分后运行
"""

import os
import torch

# ==================== 配置区 ====================
CONFIG = {
    "model_name": "whisperx",           # 模型名称
    "model_root": r"E:\ai-deploy\models",
    "hf_token": os.environ.get("HF_TOKEN", ""),  # 从环境变量读取
    "hf_mirror": "https://hf-mirror.com",
}

# ==================== 初始化 ====================
os.environ["HF_ENDPOINT"] = CONFIG["hf_mirror"]
os.environ["HF_TOKEN"] = CONFIG["hf_token"]

# PyTorch weights_only 补丁（兼容性）
_original_torch_load = torch.load
def _patched_load(*args, **kwargs):
    kwargs['weights_only'] = False
    return _original_torch_load(*args, **kwargs)
torch.load = _patched_load

# ==================== 模型下载函数 ====================
def download_whisperx():
    """下载 WhisperX 模型"""
    import whisperx

    print("下载 Whisper large-v3 模型...")
    model = whisperx.load_model(
        "large-v3",
        device="cuda",
        compute_type="float16",
        download_root=CONFIG["model_root"]
    )

    print("下载中文对齐模型...")
    align_model, metadata = whisperx.load_align_model(
        language_code="zh",
        device="cuda",
        model_dir=CONFIG["model_root"]
    )

    print("WhisperX 模型下载完成！")

def download_qwen_asr():
    """下载 Qwen ASR 模型"""
    from transformers import AutoModelForCausalLM, AutoTokenizer

    model_name = "Qwen/Qwen-Audio"

    print(f"下载 {model_name} 模型...")
    tokenizer = AutoTokenizer.from_pretrained(
        model_name,
        cache_dir=CONFIG["model_root"],
        trust_remote_code=True
    )
    model = AutoModelForCausalLM.from_pretrained(
        model_name,
        cache_dir=CONFIG["model_root"],
        trust_remote_code=True,
        torch_dtype=torch.float16,
        device_map="auto"
    )

    print("Qwen ASR 模型下载完成！")

# ==================== 主函数 ====================
def main():
    print(f"开始下载模型: {CONFIG['model_name']}")
    print(f"模型目录: {CONFIG['model_root']}")
    print(f"HF 镜像: {CONFIG['hf_mirror']}")
    print("-" * 50)

    # 根据模型名称调用不同下载函数
    downloaders = {
        "whisperx": download_whisperx,
        "qwen-asr": download_qwen_asr,
    }

    downloader = downloaders.get(CONFIG["model_name"])
    if downloader:
        downloader()
    else:
        print(f"未知模型: {CONFIG['model_name']}")
        print(f"可用模型: {list(downloaders.keys())}")

if __name__ == "__main__":
    main()
```

## 四、常见问题速查表

| 问题 | 症状 | 解决方案 |
|------|------|----------|
| pip SSL 错误 | `SSLError` | 改用阿里云镜像 |
| PyTorch 版本冲突 | `torch version mismatch` | 安装 torch 2.8.0+ |
| weights_only 错误 | `Unsupported global` | patch torch.load |
| CUDA OOM | `out of memory` | 降低 compute_type |
| HF Token 错误 | `401 Unauthorized` | 检查 Token 和协议 |
| 模型下载超时 | `ReadTimeout` | 配置 HF 镜像 |
